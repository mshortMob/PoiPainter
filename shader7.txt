precision mediump float;
precision mediump int;

uniform float time;
varying vec2 vUv;
uniform sampler2D colorTexture;
uniform sampler2D colorTexture2;

void main()	{
	vec2 position = -12.0 * vUv;
	float red = abs( sin( position.x + time / .25) );
	float green = abs( sin( position.x + 3.14/2. + time / .25 ) );
	float blue = abs( sin( position.x + 3.14/5.5 + time / .25 ) );
	vec3 mask = texture2D( colorTexture, vUv ).rgb;
	gl_FragColor = vec4( red, green, blue, mask.r );
}
