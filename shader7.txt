precision mediump float;
precision mediump int;

uniform float time;

varying vec2 vUv;

void main()	{
	vec2 position = -4.0 * vUv;
	float red = abs( sin( position.x * position.y + time / .7 ) );
	float green = abs( sin( position.x * position.y + time / .4 ) );
	float blue = abs( sin( position.x * position.y + time / .9 ) );
	gl_FragColor = vec4( red, green, blue, 1.0 );
}
