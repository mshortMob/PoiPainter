precision mediump float;
precision mediump int;

uniform float time;

varying vec2 vUv;

void main()	{
	vec2 p = - 1.0 + 2.0 * vUv;

	for(int i=1;i<100;i++)
	{
		vec2 newp=p;
		float speed = 45.0; // speed control
		newp.x+=0.8/float(i)*sin(float(i)*p.y+time/(100.0/speed)+0.8*float(i))+1.0;
		newp.y+=0.8/float(i)*sin(float(i)*p.x+time/(100.0/speed)+0.3*float(i+10))-11.4;
		p=newp;
	}
	vec3 col=vec3(.77*sin(p.x*1.+p.y), 0.2*sin(12.0*p.y), .77*sin(p.x*1.));
	gl_FragColor=vec4(col, 1.0);
}
